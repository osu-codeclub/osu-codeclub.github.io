1 Introduction

Setting up

We will also use some data sets from the usdata (as in: US data) package, so go ahead and install that:

install.packages("usdata")
library(usdata)

We’ll set a ggplot theme for all the following plots:

theme_set(theme_void())
theme_update(legend.position = "top")

Grab out state and county map data:

states <- map_data(map = "state")
ohio <- map_data(map = "county") |>
  filter(region == "ohio")


2 A first choropleth map

In choropleth maps, the color of a geographic area differs based on some variable, like population density, median income, mean temperature, or biodiversity. In ggplot-speak, we would be mapping the fill aesthetic to a variable like population density.

states |>
  filter(region %in% c("ohio", "indiana", "michigan")) |> 
  ggplot() +
  geom_polygon(
    aes(x = long, y = lat, group = group, fill = region),
    color = "grey40"
    ) +
  coord_sf()

4 Countywise

county
ohio_data <- county |>
  filter(state == "Ohio") |>
  rename(county = name) |> 
  mutate(county = sub(" County", "", county))
ohio_map <- tibble(ohio) |>
  rename(county = subregion) |>
  mutate(county = tools::toTitleCase(county))
unique(ohio_map$county) %in% ohio_data$county
all(unique(ohio_map$county) %in% ohio_data$county)
all(ohio_data$county %in% unique(ohio_map$county))
ohio_mapdata <- left_join(ohio_map, ohio_data, by = "county")
ohio_mapdata

6 Interactive maps

# See https://r-graph-gallery.com/414-map-multiple-charts-in-ggiraph.html
library(ggiraph)
p <- states |>
  filter(region %in% c("ohio", "indiana", "michigan")) |> 
  ggplot() +
  geom_polygon_interactive(
    aes(x = long, y = lat, group = group, fill = region, tooltip = region),
    color = "grey40"
    ) +
  coord_sf()
girafe(ggobj = p)
